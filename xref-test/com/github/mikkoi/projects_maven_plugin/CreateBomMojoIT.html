45           * Number of projects in the test Maven build.
46           */
47          private final int NUMBER_OF_PROJECTS_IN_BUILD = 6; // Including root
48  
49          // Create bom with all the projects in the build (set-001)
50          // Default settings
51          @MavenTest
52          @Order(1)
53          void the_first_test_case(final MavenExecutionResult result) {
54              assertThat(result).isSuccessful();
55              Path bomPath = Paths.get(result.getMavenProjectResult().getTargetProjectDirectory().toString(), "target", "bom", "pom.xml");
56              assertThat(Files.exists(bomPath)).isTrue();
57  
58              final ModelReader modelReader = new DefaultModelReader();
59              Model model;
60              try {
61                  model = modelReader.read(bomPath.toFile(), null);
62              } catch (IOException e) {
63                  throw new RuntimeException(e);
64              }
65              // Number of dependencies in DependencyManagement match number of projects in the build.
66              assertThat(model.getDependencyManagement().getDependencies().size()).isEqualTo(NUMBER_OF_PROJECTS_IN_BUILD);
67  
68              // BOM project details match root project equivalents.
69              assertThat(model.getGroupId()).isEqualTo(result.getMavenProjectResult().getModel().getGroupId());
70              assertThat(model.getVersion()).isEqualTo(result.getMavenProjectResult().getModel().getVersion());
71  
72              // BOM project specific details (not copied from root project).
73              assertThat(model.getArtifactId()).isEqualTo("bom");
74              assertThat(model.getPackaging()).isEqualTo("pom");
75              assertThat(model.getName()).isEqualTo("Project BOM");
76  
77              // Check one dependency from BOM project's DependencyManagement section.
78              // The first dependency is the root project itself.
79              // (unless deactivated with parameter or with different sorting order.)
80              final Dependency rootProject = getRootProjectAsDependency(result);
81              assertThat(model.getDependencyManagement().getDependencies().get(0).getGroupId()).isEqualTo(rootProject.getGroupId());
82              assertThat(model.getDependencyManagement().getDependencies().get(0).getArtifactId()).isEqualTo(rootProject.getArtifactId());
83              assertThat(model.getDependencyManagement().getDependencies().get(0).getVersion()).isEqualTo(rootProject.getVersion());
84              assertThat(model.getDependencyManagement().getDependencies().get(0).getType()).isEqualTo(rootProject.getType());
85          }

192     private Dependency getRootProjectAsDependency(final MavenExecutionResult result) {
193         final Dependency rootProject = new Dependency();
194         rootProject.setType(result.getMavenProjectResult().getModel().getPackaging());
195         rootProject.setGroupId(result.getMavenProjectResult().getModel().getGroupId());
196         rootProject.setArtifactId(result.getMavenProjectResult().getModel().getArtifactId());
197         rootProject.setVersion(result.getMavenProjectResult().getModel().getVersion());
198         return rootProject;
199     }
