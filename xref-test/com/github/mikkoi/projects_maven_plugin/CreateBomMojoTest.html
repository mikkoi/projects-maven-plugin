30          try {
31              Files.createDirectories(path.getParent());
32              CreateBomMojo.writePOM(path, model);

34              Files.delete(path);

36              assertThat(false).isTrue().describedAs("Test fails");

40      @Test
41      public void testIsIncluded() {
42          MavenProject project = new MavenProject();
43  
44          CreateBomMojo mojo = new CreateBomMojo();
45          List<String> includes = new ArrayList<>();
46          List<String> excludes = new ArrayList<>();
47          mojo.setIncludes(includes);
48          mojo.setExcludes(excludes);
49  
50          // This is what CreateBomMojo.validateParameters would do:
51          includes.add("*");
52          project.setGroupId("com.github.mikkoi");
53          project.setArtifactId("test-artifact");
54          assertThat(mojo.isIncluded(project)).isTrue();
55  
56          includes.add("com.github.mikkoi:*");
57          excludes.add("com.github.mikkoi:test-artifact");
58  
59          project.setGroupId("com.github.mikkoi");
60          project.setArtifactId("test-artifact");
61          assertThat(mojo.isIncluded(project)).isFalse();
62  
63          project.setArtifactId("test-artifact-2");
64          assertThat(mojo.isIncluded(project)).isTrue();
65  
66          excludes.add("com.gitlab.other:other-artifact");
67          project.setArtifactId("other-artifact");
68          assertThat(mojo.isIncluded(project)).isTrue();
69  
70          project.setGroupId("com.gitlab.other");
71          assertThat(mojo.isIncluded(project)).isFalse();
72  
73          excludes.add("com.gitlab.second:*-other-artifact");
74          project.setGroupId("com.gitlab.second");
75          project.setArtifactId("diff-other-artifact");
76          assertThat(mojo.isIncluded(project)).isFalse();
77  
78          project.setArtifactId("artifact-something");
79          assertThat(mojo.isIncluded(project)).isTrue();
80      }
