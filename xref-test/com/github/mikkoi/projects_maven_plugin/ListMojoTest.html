36      @Test
37      void testFormatProject() {
38          MavenProject project = new MavenProject();
39          project.setGroupId("this.group.id-has-dashes");
40          project.setArtifactId("this-artifact-id");
41          project.setName("This Artifact Name");
42          project.setDescription("This Artifact Description\n has two rows.");
43          project.setVersion("1.2.3-DEBUG");
44          project.setPackaging("ear");
45          project.setFile(new File("/this/base/dir/pom.xml"));
46  
47          ListMojo mojo = new ListMojo();
48  
49          mojo.setPrintFormat("{groupId}:{artifactId}:{name}");
50          assertThat(mojo.formatProject(project)).isEqualTo("this.group.id-has-dashes:this-artifact-id:This Artifact Name");
51  
52          mojo.setPrintFormat("{artifactId}:{packaging}:{version}:{description}:{absPath}");
53          assertThat(mojo.formatProject(project)).isEqualTo("this-artifact-id:ear:1.2.3-DEBUG:This Artifact Description\n has two rows.:/this/base/dir");
