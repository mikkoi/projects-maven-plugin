16      @Test
17      public void testConvertStringForMatching() {
18          String name1 = MojoUtilities.convertStringForMatching("apache");
19          assertThat(name1).isEqualTo(".*:apache:.*");
20  
21          String name2 = MojoUtilities.convertStringForMatching("org.apache.maven:core:jar");
22          assertThat(name2).isEqualTo("org\\.apache\\.maven:core:jar");
23  
24          String name3 = MojoUtilities.convertStringForMatching("org.apache.maven:core");
25          assertThat(name3).isEqualTo("org\\.apache\\.maven:core:.*");
26  
27          String name4 = MojoUtilities.convertStringForMatching("org.apache.*:*");
28          assertThat(name4).isEqualTo("org\\.apache\\..*:.*:.*");
29  
30          String name5 = MojoUtilities.convertStringForMatching("com.github.mikkoi:test-artifact");
31          assertThat(name5).isEqualTo("com\\.github\\.mikkoi:test-artifact:.*");
32  
33      }

35      @Test
36      public void testIsIncluded() {
37          MavenProject project = new MavenProject();
38  
39          List<String> includes = new ArrayList<>();
40          List<String> excludes = new ArrayList<>();
41  
42          // This is what CreateBomMojo.validateParameters would do:
43          includes.add("*");
44          project.setGroupId("com.github.mikkoi");
45          project.setArtifactId("test-artifact");
46          project.setPackaging("jar");
47          assertThat(MojoUtilities.isIncluded(includes, excludes, project)).isTrue();
48  
49          includes.add("com.github.mikkoi:*");
50          excludes.add("com.github.mikkoi:test-artifact");
51  
52          project.setGroupId("com.github.mikkoi");
53          project.setArtifactId("test-artifact");
54          assertThat(MojoUtilities.isIncluded(includes, excludes, project)).isFalse();
55  
56          project.setArtifactId("test-artifact-2");
57          assertThat(MojoUtilities.isIncluded(includes, excludes, project)).isTrue();
58  
59          excludes.add("com.gitlab.other:other-artifact");
60          project.setArtifactId("other-artifact");
61          assertThat(MojoUtilities.isIncluded(includes, excludes, project)).isTrue();
62  
63          project.setGroupId("com.gitlab.other");
64          assertThat(MojoUtilities.isIncluded(includes, excludes, project)).isFalse();
65  
66          excludes.add("com.gitlab.second:*-other-artifact");
67          project.setGroupId("com.gitlab.second");
68          project.setArtifactId("diff-other-artifact");
69          assertThat(MojoUtilities.isIncluded(includes, excludes, project)).isFalse();
70  
71          project.setArtifactId("artifact-something");
72          assertThat(MojoUtilities.isIncluded(includes, excludes, project)).isTrue();
73      }
