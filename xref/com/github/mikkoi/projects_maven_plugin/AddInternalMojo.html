77      /**
78       * Add internal projects to the current project's dependencies.
79       *
80       * @param projects      List of MavenProject objects.
81       * @return strings      List of strings ready for writing out.
82       */
83      public List<String> addInternal(List<MavenProject> projects) {
84          getLog().debug("Begin of projects:add-internal");
85          List<String> rows = new ArrayList<>();
86          MavenProject currentProject = this.mavenSession.getCurrentProject();
87          getLog().debug(String.format("Current Project: %s:%s", currentProject.getGroupId(), currentProject.getArtifactId()));
88  
89          getLog().debug("Iterate through all projects in Maven Dependency Graph, i.e. the build.");
90          mavenSession.getProjectDependencyGraph().getSortedProjects().forEach(project -> {
91              getLog().debug(String.format("    %s:%s:%s",
92                      project.getGroupId(),
93                      project.getArtifactId(),
94                      project.getVersion()
95              ));
96              if(!project.getArtifactId().equals(currentProject.getArtifactId()) && isIncluded(project)) {
97                  Dependency dependency = new Dependency();
98                  dependency.setGroupId(project.getGroupId());
99                  dependency.setArtifactId(project.getArtifactId());
100                 dependency.setVersion(project.getVersion());
101                 dependency.setScope("compile");
102                 @SuppressWarnings("unchecked") List<Dependency> currentProjectDependencies = currentProject.getDependencies();
103                 currentProjectDependencies.add(dependency);
104                 getLog().info(String.format("Add dependency %s:%s:%s to project %s",
105                         project.getGroupId(),
106                         project.getArtifactId(),
107                         project.getVersion(),
108                         currentProject.getId()
109                 ));
110             }
111         });
112         getLog().debug("End of iterate");
113 
114         getLog().debug(":End of projects:add-internal");
115         return rows;
116     }

118     /**
119      * Create a comparator which compares Maven project groupId and artifactId
120      * for alphabetical listing.
121      *
122      * @param sortOrder "random" / "alphabetic"
123      * @return the comparator
124      */
125     static Comparator<MavenProject> getMavenProjectComparator(String sortOrder) {
126         Comparator<MavenProject> comparator;
127         if ("maven".equals(sortOrder)) {
128             comparator = new Comparator<MavenProject>() {
129                 @Override
130                 public int compare(MavenProject o1, MavenProject o2) {
131                     return 0;
132                 }
133             };
134         } else {
135             comparator = (o1, o2) -> {
136                 String o1GroupId = o1.getGroupId();
137                 String o2groupId = o2.getGroupId();
138                 String o1artifactId = o1.getArtifactId();
139                 String o2artifactId = o2.getArtifactId();
140                 int r = o1GroupId.compareTo(o2groupId);
141                 if (r != 0) {
142                     return r;
143                 }
144                 return o1artifactId.compareTo(o2artifactId);
145             };
146         }
147         return comparator;
148     }

150     /**
151      * Convert string for matching.
152      * Prepare string for use with Java's regexp libraries.
153      *
154      * @param s String
155      * @return converted string
156      */
157     static String convertStringForMatching(String s) {
158         String t = s.replace(".", "\\.");
159         t = t.replace("*", ".*");
160         if (!t.contains(":")) {
161             t = ".*:" + t;
162         }
163         return t;
164     }

166     /**
167      * <p>Replace a String with another String inside a larger String,
168      * for the first <code>max</code> values of the search String.</p>
169      *
170      * <p>A <code>null</code> reference passed to this method is a no-op.</p>
171      *
172      * <p>Copied and modified from org.apache.maven.shared.utils.StringUtils.</p>
173      *
174      * @param text text to search and replace in
175      * @param repl String to search for
176      * @param with String to replace with
177      * @return the text with any replacements processed
178      */
179     static String replace(String text, String repl, String with) {
180         if ((text == null) || (repl == null) || (with == null) || repl.isEmpty()) {
181             return text;
182         }
183 
184         StringBuilder buf = new StringBuilder(text.length());
185         int start = 0, end;
186         while ((end = text.indexOf(repl, start)) != -1) {
187             buf.append(text, start, end).append(with);
188             start = end + repl.length();
189         }
190         buf.append(text, start, text.length());
191         return buf.toString();
192     }

194     /**
195      * Decide if the project is included or excluded.
196      *
197      * @param mavenProject MavenProject
198      * @return true if included, false if excluded
199      */
200     public boolean isIncluded(MavenProject mavenProject) {
201         String projectGroupId = mavenProject.getGroupId();
202         String projectArtifactId = mavenProject.getArtifactId();
203         String projectId = projectGroupId + ":" + projectArtifactId;
204 
205         // Match from the end of the id, artifactId alone is enough.
206         Predicate<String> predicateForProjectId = s -> projectId.matches(convertStringForMatching(s));
207         if (this.includes.isEmpty() && this.excludes.isEmpty()) {
208             return true;
209         } else if (this.includes.isEmpty()) {  // && ! this.excludes.isEmpty()
210             return this.excludes.stream().noneMatch(predicateForProjectId);
211         } else if (this.excludes.isEmpty()) {  // && !this.includes.isEmpty()
212             return this.includes.stream().anyMatch(predicateForProjectId);
213         } else {
214             return this.includes.stream().anyMatch(predicateForProjectId) && this.excludes.stream().noneMatch(predicateForProjectId);
215         }
216     }
