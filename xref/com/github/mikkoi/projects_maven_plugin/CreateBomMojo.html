125     /**
126      * Validate parameters provided via properties
127      * either on the command line or using configuration element in pom.
128      */
129     private void validateAndPrepareParameters() throws MojoExecutionException {
130         getLog().debug("includes=" + includes);
131         getLog().debug("excludes=" + excludes);
132         getLog().debug("sortOrder=" + sortOrder);
133         getLog().debug("bomFilepath=" + bomFilepath);
134         getLog().debug("bomGroupId=" + bomGroupId);
135         getLog().debug("bomArtifactId=" + bomArtifactId);
136         getLog().debug("bomVersion=" + bomVersion);
137         getLog().debug("bomName=" + bomName);
138 
139         for ( String a : includes ) {
140             if (a == null) {
141                 throw new MojoExecutionException(includes, "Failure in parameter", "Failure in parameter 'includes'. String is null");
142             }
143         }
144         if (includes.isEmpty()) {
145             includes.add("*");
146         }
147 
148         for ( String a : excludes ) {
149             if (a == null) {
150                 throw new MojoExecutionException(excludes, "Failure in parameter", "Failure in parameter 'excludes'. String is null");
151             }
152         }
153 
154         if(! ("maven".equals(sortOrder) || "alphabetical".equals(sortOrder))) {
155             throw new MojoExecutionException(sortOrder, "Failure in parameter", "Failure in parameter 'sortOrder'. Allowed values: 'maven', 'alphabetical'.");
156         }
157 
158         if("DEFAULT_TO_BE_REPLACED".equals(bomFilepath)) {
159             // Replace with default: target/bom/pom.xml
160             bomFilepath = mavenSession.getCurrentProject().getBuild().getDirectory() + "/bom/pom.xml";
161         } else if (bomFilepath.isEmpty()) {
162             throw new MojoExecutionException(bomFilepath, "Failure in parameter", "Failure in parameter 'bomPath'. String is null");
163         }
164 
165         // Validate bomPath
166         try {
167             Path pathParent = Paths.get(bomFilepath).getParent();
168             getLog().debug("pathParent=" + pathParent);
169             if(pathParent != null) {
170                 Files.createDirectories(pathParent);
171             }
172         } catch (IOException e) {
173             throw new MojoExecutionException(
174                     bomFilepath,
175                     "Failure in parameter",
176                     String.format("Failure in parameter 'bomPath'. Cannot create path '%s'", Paths.get(bomFilepath))
177             );
178         }
179 
180         if("DEFAULT_TO_BE_REPLACED".equals(bomGroupId)) {
181             bomGroupId = mavenSession.getCurrentProject().getGroupId();
182         }
183 
184         if("DEFAULT_TO_BE_REPLACED".equals(bomVersion)) {
185             bomVersion = mavenSession.getCurrentProject().getVersion();
186         }
187 
188         getLog().debug("includes(resolved)=" + includes);
189         getLog().debug("excludes(resolved)=" + excludes);
190         getLog().debug("sortOrder(resolved)=" + sortOrder);
191         getLog().debug("bomFilepath(resolved)=" + bomFilepath);
192         getLog().debug("bomGroupId(resolved)=" + bomGroupId);
193         getLog().debug("bomArtifactId(resolved)=" + bomArtifactId);
194         getLog().debug("bomVersion(resolved)=" + bomVersion);
195         getLog().debug("bomName(resolved)=" + bomName);
196     }

198     /**
199      * Create a comparator which compares Maven project groupId, artifactId and type
200      * for alphabetical listing.
201      *
202      * @param sortOrder "random" / "alphabetic"
203      * @return the comparator
204      */
205     static Comparator<MavenProject> getMavenProjectComparator(String sortOrder) {
206         Comparator<MavenProject> comparator;
207         if ("maven".equals(sortOrder)) {
208             comparator = (o1, o2) -> 0;
209         } else {
210             comparator = (o1, o2) -> {
211                 String o1GroupId = o1.getGroupId();
212                 String o2groupId = o2.getGroupId();
213                 String o1artifactId = o1.getArtifactId();
214                 String o2artifactId = o2.getArtifactId();
215                 String o1type = o1.getPackaging();
216                 String o2type = o2.getPackaging();
217                 int r = o1GroupId.compareTo(o2groupId);
218                 if (r != 0) {
219                     return r;
220                 }
221                 int s = o1artifactId.compareTo(o2artifactId);
222                 if (s != 0) {
223                     return s;
224                 }
225                 return o1type.compareTo(o2type);
226             };
227         }
228         return comparator;
229     }

231     /**
232      * Write a POM.
233      *
234      * @param path  String, path to file.
235      * @param model Maven POM {@link org.apache.maven.model.Model}.
236      */
237     static void writePOM(Path path, Model model) throws IOException {
238         final ModelWriter modelWriter = new DefaultModelWriter();
239         try (OutputStream outputStream = Files.newOutputStream(path)) {
240             modelWriter.write(outputStream, null, model);
241         }

244     /**
245      * Convert list of projects to a list of strings.
246      *
247      * @param mavenSession MavenSession
248      */
249     void createBom(MavenSession mavenSession) {
250         getLog().debug(String.format("Begin of createBom(%s)", mavenSession));
251         final MavenProject currentProject = mavenSession.getCurrentProject();
252 
253         // Create a new model so we don't pollute the current one.
254         final Model model = currentProject.getModel().clone();
255         getLog().debug(String.format("model=%s", model));
256 
257         // Remove from model the following elements:
258         model.setBuild(null);
259         model.setDependencies(null);
260         model.setPluginRepositories(null);
261         model.getDependencies().clear();
262         model.getProfiles().clear();
263         model.setPrerequisites(null);
264         model.setRepositories(null);
265         model.setProperties(null);
266         model.setReporting(null);
267         model.setParent(null);
268         model.setModules(null);
269 
270         model.setGroupId(this.bomGroupId);
271         model.setArtifactId(this.bomArtifactId);
272         model.setVersion(this.bomVersion);
273         model.setName(this.bomName);
274 
275         // Clear dependencyManagement and fill it with all projects in the build.
276         getLog().debug(String.format("model.getDependencyManagement=%s", model.getDependencyManagement()));
277         model.setDependencyManagement(new DependencyManagement());
278 
279         if(includeProjects) {
280             final List<MavenProject> projects = mavenSession.getProjects();
281             Comparator<MavenProject> comparator = getMavenProjectComparator(this.sortOrder);
282             projects.stream().filter(this::isIncluded).sorted(comparator).forEach(mavenProject -> {
283                 final Dependency dependency = new Dependency();
284                 dependency.setGroupId(mavenProject.getGroupId());
285                 dependency.setArtifactId(mavenProject.getArtifactId());
286                 dependency.setVersion(mavenProject.getVersion());
287                 dependency.setType(mavenProject.getArtifact().getType());
288                 model.getDependencyManagement().addDependency(dependency);
289             });
290         }
291 
292         // Save the BOM POM
293         try {
294             writePOM(Paths.get(this.bomFilepath).toAbsolutePath(), model);
295             getLog().info(String.format("BOM saved in %s", this.bomFilepath));
296         } catch (IOException e) {
297             getLog().error(String.format("Cannot write POM %s", this.bomFilepath), e);
298         }
299 
300         if(attachToCurrentProject) {
301             final Model currentModel = currentProject.getOriginalModel();
302             final List<String> modules = currentModel.getModules();
303             modules.add(this.bomFilepath);
304             java.io.File modelFile = currentModel.getPomFile();
305             java.nio.file.Path p = Paths.get(modelFile.getAbsolutePath());
306             try {
307                 writePOM(p, currentModel);
308             } catch (IOException e) {
309                 getLog().error(String.format("Cannot write POM %s", p), e);
310             }
311         }
312         getLog().debug(":End of createBom");
313     }

315     /**
316      * Is the project included based on includes and excludes lists.
317      *
318      * @param mavenProject MavenProject
319      * @return true if included, false if excluded
320      */
321     public boolean isIncluded(MavenProject mavenProject) {
322         boolean r = MojoUtilities.isIncluded(this.includes, this.excludes, mavenProject);
323         getLog().debug(String.format("isIncluded(%s:%s:%s:%s): %b", mavenProject.getGroupId(), mavenProject.getArtifactId(), mavenProject.getVersion(), mavenProject.getPackaging(), r));
324         return r;
325     }

327     /**
328      * The main entry point for mojo.
329      */
330     @Override
331     public void execute() throws MojoExecutionException {
332         if (skip || thisMojoSkip) {
333             getLog().info("Skip execution ...");
334             return;
335         }
336         MavenProject currentProject = this.mavenSession.getCurrentProject();
337         getLog().debug(String.format("Current Project: %s:%s", currentProject.getGroupId(), currentProject.getArtifactId()));
338         MavenProject topLevelProject = this.mavenSession.getTopLevelProject();
339         getLog().debug(String.format("Top Level Project: %s:%s", topLevelProject.getGroupId(), topLevelProject.getArtifactId()));
340 
341         validateAndPrepareParameters();
342 
343         createBom(mavenSession);
344     }
