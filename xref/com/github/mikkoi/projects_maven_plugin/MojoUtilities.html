15      /**
16       * Create a comparator which compares Maven project groupId and artifactId
17       * for alphabetical listing.
18       *
19       * @param sortOrder "random" / "alphabetic"
20       * @return the comparator
21       */
22      static Comparator<MavenProject> getMavenProjectComparator(String sortOrder) {
23          Comparator<MavenProject> comparator;
24          if ("maven".equals(sortOrder)) {
25              comparator = (o1, o2) -> 0;
26          } else {
27              comparator = (o1, o2) -> {
28                  String o1GroupId = o1.getGroupId();
29                  String o2groupId = o2.getGroupId();
30                  String o1artifactId = o1.getArtifactId();
31                  String o2artifactId = o2.getArtifactId();
32                  String o1Packaging = o1.getPackaging();
33                  String o2Packaging = o2.getPackaging();
34                  int r1 = o1GroupId.compareTo(o2groupId);
35                  if (r1 != 0) {
36                      return r1;
37                  }
38                  int r2 = o1artifactId.compareTo(o2artifactId);
39                  if (r2 != 0) {
40                      return r2;
41                  }
42                  return o1Packaging.compareTo(o2Packaging);
43              };
44          }
45          return comparator;
46      }

48      /**
49       * Convert string for matching.
50       *
51       * @param s String
52       * @return converted string
53       */
54      public static String convertStringForMatching(String s) {
55          String t = s.replace(".", "\\.");
56          t = t.replace("*", ".*");
57          if (!t.contains(":")) {
58              t = ".*:" + t + ":.*";
59          }
60          if (Arrays.stream(t.split(":")).count() < 3) {
61              t = t + ":.*";
62          }
63          return t;
64      }

66      /**
67       * Is the project included based on includes and excludes lists.
68       * <p>
69       * The includes and excludes lists are lists of strings which are
70       * matched against the project id, which is groupId:artifactId:packaging.
71       * <p>
72       * The matching is done from the end of the string, so artifactId alone is enough.
73       * <p>
74       * The includes and excludes lists are processed in order, so the first match wins.
75       * <p>
76       * If includes is empty, then all projects are included.
77       * If excludes is empty, then no projects are excluded.
78       *
79       * @param includes      List of strings (project ids) to include
80       * @param excludes      List of strings (project ids) to exclude
81       * @param mavenProject MavenProject to check
82       * @return true if the project is included
83       */
84      public static boolean isIncluded(List<String> includes, List<String> excludes, MavenProject mavenProject) {
85          String projectId = String.format("%s:%s:%s", mavenProject.getGroupId(), mavenProject.getArtifactId(), mavenProject.getPackaging());
86  
87          // Match from the end of the id, artifactId alone is enough.
88          Predicate<String> predicateForProjectId = s -> projectId.matches(convertStringForMatching(s));
89          return includes.stream().anyMatch(predicateForProjectId) && excludes.stream().noneMatch(predicateForProjectId);
90      }
